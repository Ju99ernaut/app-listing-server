//// -- Tables and References

// Creating tables
Table users as U {
  id int [pk, increment] // auto-increment
  username varchar [not null, unique]
  email varchar [not null, unique]
  password varchar
  joined datetime
  admin bool
}

Table applications {
  id int [pk, increment] // auto-increment
  image varchar
  title varchar [not null, unique]
  by varchar
  groups [varchar]
  description varchar
  updated datetime
  owner varchar [ref: - U.username]
}

Table ratings {
  id int [pk, increment] // auto-increment
  user varchar [ref: - U.username]
  application varchar [ref: - applications.title]
  rating float
  comment varchar
  updated datetime
}
